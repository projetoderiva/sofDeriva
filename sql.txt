-- Database: bancoderiva

-- DROP DATABASE bancoderiva;

CREATE DATABASE bancoderiva



CREATE TABLE fazenda
(
  id serial NOT NULL,
  nome character varying(100),
  registro character varying(100),
  cidade character varying(50),
  estado character varying(50),
  tamanho character varying(50),
  latitude integer,
  longitude integer,
  CONSTRAINT fazenda_pkey PRIMARY KEY (id)
)
CREATE TABLE pessoa
(
  id serial NOT NULL,
  nome character varying(50) NOT NULL,
  email character varying(50) NOT NULL,
  idade integer,
  CONSTRAINT pessoa_pkey PRIMARY KEY (id)
)
-- Table: plataforma

-- DROP TABLE plataforma;

CREATE TABLE plataforma
(
  id serial NOT NULL,
  localizacao character(1),
  data date,
  CONSTRAINT plataforma_pkey PRIMARY KEY (id)
)
-- Table: rssf

-- DROP TABLE rssf;

CREATE TABLE rssf
(
  id serial NOT NULL,
  "sensorTemperatura" numeric,
  "sensorUmidade" numeric,
  "sensorUltrassonico" numeric,
  "sensorAnemometro" numeric,
  "gpsLongitude" numeric,
  "gpsLatitude" numeric,
  "gpsVelocidade" numeric,
  "gpsData" date,
  "gpsHora" numeric,
  CONSTRAINT rssf_pkey PRIMARY KEY (id)
)

-- Table: talhao

-- DROP TABLE talhao;

CREATE TABLE talhao
(
  id serial NOT NULL,
  nome character varying(50),
  localizacao character varying(30),
  tipo character varying(20),
  CONSTRAINT talhao_pkey PRIMARY KEY (id)
)

-- Table: usuario

-- DROP TABLE usuario;

CREATE TABLE usuario
(
  id serial NOT NULL,
  nome character varying(50),
  email character varying(30),
  senha character varying(20),
  CONSTRAINT usuario_pkey PRIMARY KEY (id)
)
